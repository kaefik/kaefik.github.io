Установка Docker в Ubuntu

 sudo apt-get update
 sudo apt-get install \
    apt-transport-https \
    ca-certificates \
    curl \
    gnupg \
    lsb-release


 curl -fsSL https://download.docker.com/linux/ubuntu/gpg | sudo gpg --dearmor -o /usr/share/keyrings/docker-archive-keyring.gpg


 echo \
  "deb [arch=$(dpkg --print-architecture) signed-by=/usr/share/keyrings/docker-archive-keyring.gpg] https://download.docker.com/linux/ubuntu \
  $(lsb_release -cs) stable" | sudo tee /etc/apt/sources.list.d/docker.list > /dev/null

sudo apt-get update
 sudo apt-get install docker-ce docker-ce-cli containerd.io
sudo systemctl status docker


# добавление текущего пользователя в группу docker
 sudo usermod -aG docker $USER

Установка docker в Fedora 27

sudo dnf -y install dnf-plugins-core
sudo dnf config-manager \
  --add-repo \
  https://download.docker.com/linux/fedora/docker-ce.repo
sudo dnf config-manager --set-enabled docker-ce-edge
sudo dnf install docker
service docker start

Возможные проблемы

Ошибка при запуске docker: Cannot connect to the Docker daemon at unix:///var/run/docker.sock. Is the docker daemon running?

запустить процесс: Once Docker is installed, you will need to start the Docker daemon.
